\title{Fun with Multiple Choice}
\author{Jim Fowler}

\begin{document}
\begin{abstract}
    Demonstrating a variety of super-fun multiple choice questions in Ximera.
\end{abstract}
\maketitle

\begin{exercise}
    Using Ximera feels like:
    \begin{multipleChoice}
        \choice{Riding a unicorn through a cotton‑candy cloud}
        \choice{Sipping a milkshake while skydiving}
        \choice[correct]{Playing laser tag in zero‑gravity}
        \choice{Tickling a giant, friendly octopus}
        \choice{Dancing the tango with a robot DJ}
    \end{multipleChoice}
\end{exercise}
